function setCookie(name, value, days) {
const d = new Date();
d.setTime(d.getTime() + (days * 24 * 60 * 60 * 1000));
document.cookie = `${name}=${value};expires=${d.toUTCString()};path=/;SameSite=Lax`;
}

function getCookie(name) {
const v = document.cookie.match('(^|;)\\s*' + name + '\\s*=\\s*([^;]+)');
return v ? v.pop() : null;
}

function abrirIDB() {
return new Promise((res, rej) => {
const req = indexedDB.open('turnos2026', 1);
req.onupgradeneeded = e => e.target.result.createObjectStore('sesion', { keyPath: 'k' });
req.onsuccess = e => { idbDB = e.target.result; res(idbDB); };
req.onerror = () => rej();
});
}

function idbSet(key, value) {
return abrirIDB().then(db => new Promise((res, rej) => {
const tx = db.transaction('sesion', 'readwrite');
tx.objectStore('sesion').put({ k: key, v: value });
tx.oncomplete = res; tx.onerror = rej;
}));
}

function idbGet(key) {
return abrirIDB().then(db => new Promise((res) => {
const tx = db.transaction('sesion', 'readonly');
const req = tx.objectStore('sesion').get(key);
req.onsuccess = () => res(req.result ? req.result.v : null);
req.onerror = () => res(null);
}));
}

function restaurarSesion() {
const misGrupos = cargarMisGrupos();

// Si hay grupos guardados, mostrar pantalla de selección
if (misGrupos.length > 0) {
mostrarPantallaGrupos();
return true;
}

// Si no hay grupos, intentar sesión individual guardada
try {
const raw = localStorage.getItem('sesion2026');
if (raw) {
const s = JSON.parse(raw);
if (s && s.miID && !s.modoGrupo) { aplicarSesion(s); return true; }
}
} catch(e) {}

// 2. Intentar IndexedDB (iOS Safari standalone)
idbGet('sesion2026').then(raw => {
if (raw) {
try {
const s = JSON.parse(raw);
if (s && s.miID) { aplicarSesion(s); return; }
} catch(e) {}
}
// 3. Intentar cookie
const cookieID = getCookie('uid2026');
if (cookieID) {
db.ref(`sesiones/${cookieID}`).once('value').then(snap => {
const s = snap.val();
if (s && s.miID) { aplicarSesion(s); }
else { rellenarCamposLogin(); }
});
} else {
rellenarCamposLogin();
}
}).catch(() => rellenarCamposLogin());

return true;
}

function guardarEdicionMiembro(oldId) {
const nom = document.getElementById('ei-nom-' + oldId).value.trim();
const newId = document.getElementById('ei-id-' + oldId).value.trim();
const tipo = document.getElementById('ei-tipo-' + oldId).value;
const grupo = document.getElementById('ei-grp-' + oldId).value;

const updates = {};
// Si cambió el ID: borrar el nodo viejo y crear uno nuevo
if (newId !== oldId) {
if (newId === miID) { alert("Ese ID ya eres tú"); return; }
updates[`grupos/${grupoActual}/miembros/${oldId}`] = null;
}
updates[`grupos/${grupoActual}/miembros/${newId}`] = { n: nom, id: newId, g: grupo, t: tipo };
// Si el miembro editado somos nosotros mismos, actualizar sesión
if (oldId === miID) {
miID = newId; miNombreCompleto = nom; miTipoSesion = tipo; miGrupoSesion = grupo;
guardarSesion();
}
db.ref().update(updates).then(() => abrirEditorGrupo());
}

}

function iniciarDragMenu(menu, handle) {
let startX, startY, startLeft, startTop, dragging = false;

function onDown(e) {
dragging = true;

startX = t.clientX; startY = t.clientY;

e.preventDefault();
}
function onMove(e) {
if (!dragging) return;
const t = e.touches ? e.touches[0] : e;
const dx = t.clientX - startX, dy = t.clientY - startY;
let newLeft = startLeft + dx, newTop = startTop + dy;
newLeft = Math.max(0, Math.min(newLeft, window.innerWidth - 250));
newTop = Math.max(0, Math.min(newTop, window.innerHeight - 50));
menu.style.left = newLeft + 'px';
menu.style.top = newTop + 'px';
e.preventDefault();
}
function onUp() { dragging = false; }

handle.addEventListener('touchstart', onDown, { passive: false });
document.addEventListener('mousemove', onMove);

cerrarDesplegables();

}

if (esVacaciones) {
const totalDias = new Date(2026, mIdx + 1, 0).getDate();
const diasVac = [];
for (let i = 1; i <= totalDias; i++) {

function enviarMensaje() {
const input = document.getElementById('chat-input');
const texto = input.value.trim();
if (!texto && !archivoBase64Pendiente) return;
if (!grupoActual) { alert('No estás en un grupo'); return; }
const nombre = miNombreCompleto ? miNombreCompleto.split(' ')[0] : ('ID ' + miID);
const msg = { uid: miID, nom: nombre, texto: texto || '', ts: Date.now() };
if (archivoBase64Pendiente) msg.imagen = archivoBase64Pendiente;
input.value = '';
cancelarArchivo();
db.ref('chats/' + grupoActual).push(msg);
}

}
});
}

function reproducirSonido() {
try {
const ctx = new (window.AudioContext || window.webkitAudioContext)();
const osc = ctx.createOscillator();
const gain = ctx.createGain();
osc.connect(gain); gain.connect(ctx.destination);
osc.frequency.setValueAtTime(880, ctx.currentTime);
osc.frequency.setValueAtTime(660, ctx.currentTime + 0.1);
gain.gain.setValueAtTime(0.3, ctx.currentTime);
gain.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.4);
osc.start(ctx.currentTime);
osc.stop(ctx.currentTime + 0.4);
} catch(e) {}
}

function renderEstadisticas() {
// Leer valores frescos de sesión
let gSesion = miGrupoSesion;
let tSesion = miTipoSesion;
// Si están vacíos intentar recuperar de Firebase sesión
if (!gSesion || !tSesion) {
try {
const raw = localStorage.getItem('sesion2026');
if (raw) {
const s = JSON.parse(raw);
gSesion = s.miGrupoSesion || '';
tSesion = s.miTipoSesion || '';
}
} catch(e) {}
}
const tabla = tSesion === '80/20' ? gruposLibres8020 : gruposLibres;
const tablaGrupo = tabla[gSesion] || tabla[parseInt(gSesion)] || null;
if (!tablaGrupo) {
document.getElementById('stats-content').innerHTML = '<p style="text-align:center; color:#94a3b8; padding:20px;">No hay datos de libres para tu grupo.<br><small>Grupo: ' + gSesion + ' | Turno: ' + (tSesion || 'no definido') + '</small></p>';
return;
}
// La tabla es un objeto {0:[...], 1:[...], ...}, convertir a array de 12 meses
const diasLibresPorMes = [];
for (let i = 0; i < 12; i++) {
diasLibresPorMes.push(tablaGrupo[i] || []);
}
const festivos2026 = ['0_1','0_6','3_2','3_9','4_1','4_15','7_15','9_12','10_1','10_9','11_6','11_8','11_25'];
const meses = ['Ene','Feb','Mar','Abr','May','Jun','Jul','Ago','Sep','Oct','Nov','Dic'];
let totalLibres = 0, finesLibres = 0, festivosLibres = 0, htmlMeses = '';

diasLibresPorMes.forEach((dias, mIdx) => {
if (!dias || dias.length === 0) return;
let fdsEsMes = 0, festEsMes = 0;
dias.forEach(d => {
const dow = new Date(2026, mIdx, d).getDay();
if (dow === 0 || dow === 6) fdsEsMes++;
if (festivos2026.includes(mIdx + '_' + d)) festEsMes++;
});
totalLibres += dias.length;
finesLibres += fdsEsMes;
festivosLibres += festEsMes;
const cambiosMiID = dbCambios[miID] || {};
const keysDelMes = Object.keys(cambiosMiID).filter(k => k.startsWith(mIdx + '_'));
let nComp = 0, nPropio = 0, nPend = 0, nGen = 0;
keysDelMes.forEach(k => {
const c = cambiosMiID[k];
if (c.clase === 'b-inter' || c.clase === 'b-cambio-libre') nComp++;
else if (c.clase === 'b-propio' || c.clase === 'b-cambio') nPropio++;
else if (c.clase === 'b-pend') nPend++;
else if (c.clase === 'b-generado') nGen++;
});
const totalCambiosMes = nComp + nPropio + nPend + nGen;
let etiquetasCambios = '';
if (nComp > 0) etiquetasCambios += '<span style="font-size:0.7rem; background:#ede9fe; color:#7c3aed; padding:2px 5px; border-radius:6px; margin-left:2px;">🤝' + nComp + '</span>';
if (nPropio > 0) etiquetasCambios += '<span style="font-size:0.7rem; background:#e0f2fe; color:#0369a1; padding:2px 5px; border-radius:6px; margin-left:2px;">🔄' + nPropio + '</span>';
if (nPend > 0) etiquetasCambios += '<span style="font-size:0.7rem; background:#fef3c7; color:#b45309; padding:2px 5px; border-radius:6px; margin-left:2px;">⏳' + nPend + '</span>';
if (nGen > 0) etiquetasCambios += '<span style="font-size:0.7rem; background:#fce7f3; color:#be185d; padding:2px 5px; border-radius:6px; margin-left:2px;">✨' + nGen + '</span>';
htmlMeses += '<div style="padding:8px 0; border-bottom:1px solid #f1f5f9;">' +
'<div style="display:flex; justify-content:space-between; align-items:center;">' +
'<span style="font-weight:700; color:var(--primary-dark); width:35px;">' + meses[mIdx] + '</span>' +
'<span style="font-size:0.8rem; color:#475569;">' + dias.length + ' libres</span>' +
'<span style="font-size:0.8rem; color:#16a34a;">' + fdsEsMes + ' fds</span>' +
'<span style="font-size:0.8rem; color:#dc2626;">' + festEsMes + ' fest</span>' +
'</div>' +
(totalCambiosMes > 0 ? '<div style="display:flex; flex-wrap:wrap; gap:3px; margin-top:4px; padding-left:35px;">' + etiquetasCambios + '</div>' : '') +
'</div>';
});

document.getElementById('stats-content').innerHTML =
'<div style="display:grid; grid-template-columns:1fr 1fr 1fr; gap:10px; margin-bottom:16px;">' +
'<div style="background:#f5f3ff; border-radius:12px; padding:12px; text-align:center;"><div style="font-size:1.8rem; font-weight:900; color:var(--primary);">' + totalLibres + '</div><div style="font-size:0.7rem; color:#64748b; font-weight:600;">TOTAL LIBRES</div></div>' +
'<div style="background:#f0fdf4; border-radius:12px; padding:12px; text-align:center;"><div style="font-size:1.8rem; font-weight:900; color:#16a34a;">' + finesLibres + '</div><div style="font-size:0.7rem; color:#64748b; font-weight:600;">FIN DE SEMANA</div></div>' +
'<div style="background:#fef2f2; border-radius:12px; padding:12px; text-align:center;"><div style="font-size:1.8rem; font-weight:900; color:#dc2626;">' + festivosLibres + '</div><div style="font-size:0.7rem; color:#64748b; font-weight:600;">FESTIVOS</div></div>' +
'</div>' +
'<div style="font-weight:700; color:var(--primary-dark); margin-bottom:8px; font-size:0.85rem;">Detalle por mes:</div>' +
'<div style="display:flex; justify-content:flex-end; gap:16px; margin-bottom:6px; font-size:0.7rem; color:#94a3b8; font-weight:600;"><span>LIBRES</span><span style="color:#16a34a">FDS</span><span style="color:#dc2626">FEST</span></div>' +
